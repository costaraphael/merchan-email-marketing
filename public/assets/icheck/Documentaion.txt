Include in head: <link href="assets/icheck/skins/all.css" rel="stylesheet">

Include in body: <script src="assets/icheck/js/jquery.icheck.min.js"></script>

Simple Usage:

<input tabindex="5" type="checkbox" class="skin-minimal">

To run:

$('.skin-minimal').iCheck({
	checkboxClass: 'icheckbox_minimal',
	radioClass: 'iradio_minimal',
	increaseArea: '20%'
});
